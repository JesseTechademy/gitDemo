git init 
    ->initialised an empty git repository

git add filename / git add . 
    ->used to add files to the staging area

git status 
    ->used to find the state of the current working directory and 
    staging area

git reset 
    ->used to remove files from the staging area

git commit -m "message"
    ->used to save the current staged changes in the project to the git

git log
    ->gives detailed log of all previous commits
git log --oneline
    ->gives simplified log of all previous commits in one line

git remote add origin https://github.com/JesseTechademy/gitDemo.git
